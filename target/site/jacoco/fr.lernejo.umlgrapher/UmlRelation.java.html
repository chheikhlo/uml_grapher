package fr.lernejo.umlgrapher;

import java.util.Set;

public class UmlRelation {
    private final Set<UmlType> Tree;

    public UmlRelation(Set<UmlType> arbre) {
        this.Tree = arbre;
    }

    public Set<Relation> Relation(Set<Relation> myRelation) {
        String link = "";
        for (UmlType i : Tree) {
            Class[] j = i.getaClasse().getInterfaces();
            for (Class k : j) {
                if (i.getaClasse().getSuperclass() == null) {
                    link = k.getSimpleName() + " <|-- " + i.getaClasse().getSimpleName() + " : extends";
                } else {
                    link = k.getSimpleName() + " <|.. " + i.getaClasse().getSimpleName() + " : implements";
                }
                link += "\n";
                myRelation.add(new Relation(k, i.getaClasse(), link));
            }
        }
        return myRelation;
    }

    public String allRelation(Set<Relation> links, Set<UmlType> types) {
        String link = "";
        links = new UmlRelation(types).Relation(links);
        for (Relation i : links) {
            link += i.getLink();
        }
        return link;
    }

}
